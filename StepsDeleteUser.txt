	@Override
	public void deleteUserByID(User u) {
		try {
			Query query = entityManager.createQuery("delete from User u where u.id = :uid");
			query.setParameter("uid", u.getId());
			query.executeUpdate();
			entityManager.flush();
		} catch (Exception e) {
			System.out.println(e.getMessage());
			return;
		}
	}
